This writes the full frame to the // socket, waiting if necessary. self.connection.write_frame(&frame).await?; // Wait for the response from the server // // Both `Simple` and `Bulk` frames are accepted. `Null` represents the // key not being present and `None` is returned. match self.read_response().await? { Frame::Simple(value) => Ok(Some(value.into())), Frame::Bulk(value) => Ok(Some(value)), Frame::Null => Ok(None), frame => Err(frame.to_error()), } }
